from pathlib import Path import re src = Path("/mnt/data/simulador_cdb_legenda_corrigida_v3.html") dst = Path("/mnt/data/simulador_cdb_corrigido_final_v4.html") html = src.read_text(encoding="utf-8") # Corrige os dois gráficos de barra: # Eles devem usar apenas as legendas manuais abaixo dos gráficos. # A legenda automática do Chart.js precisa ficar desligada para não aparecer "undefined". html = html.replace( "plugins:{ legend:{display:true, position:'top', align:'end'}, datalabels:{display:true} },", "plugins:{ legend:{display:false}, datalabels:{display:true} },", 2 ) # Garante que a legenda manual da evolução não volte a duplicar. manual_legend_pattern = r'''
\s* \s*\s* \s*A\+Reinvestimento\s* \s*\s* \s*\s* \s*Opcao B\s* \s*\s* \s*
''' html = re.sub(manual_legend_pattern, "", html, flags=re.DOTALL) # Para o gráfico de evolução, queremos manter a legenda automática do Chart.js ligada. # Procurar o bloco do chartModelBreakeven e garantir legend true apenas ali. start = html.find("chartModelBreakeven") # Primeiro occurrence é o canvas, então procurar a instância do gráfico start = html.find("window._chartModelBreakeven", start) if start != -1: end = html.find("});", start) if end != -1: block = html[start:end+3] block = re.sub( r"legend\s*:\s*\{\s*display\s*:\s*false\s*\}", "legend:{display:true, position:'top', align:'end'}", block, count=1 ) html = html[:start] + block + html[end+3:] dst.write_text(html, encoding="utf-8") print(f"Arquivo criado: {dst}")
